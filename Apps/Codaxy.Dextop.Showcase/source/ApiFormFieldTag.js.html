Ext.ns('Showcase');

Ext.define('Showcase.demos.Launcher.ApiFormFieldTag', {
    extend: 'Showcase.demos.Launcher',
    launch: function () {
        var remoteId = Ext.id();
        var w = Ext.create('Showcase.demos.ApiFormFieldTagWindow', {});
        w.show();        
    }
});

Ext.define('Showcase.demos.ApiFormFieldTagWindow', {
    extend: 'Ext.window.Window',	
    width: 500,
    border: false,
    title: 'TagField demo using Dextop API',	
	
    initComponent: function () {
	    Ext.apply(this, {
	        layout: 'fit',
	        items: [this.form = Ext.create('Ext.form.Panel', {
                reference: 'form',
                bodyPadding: 5,
                items:
                    Ext.create('Showcase.demos.form.FormFieldTag').getItems({
                    data: {
                        'CountryIds': [2, 4],
                        'CountryNames': ['Russia', 'Serbia']
                    }
                }),
                buttons: [{
                    text: 'Display value',
                    scope: this,
                    handler: function () {
                        console.log(this.form.getForm().getFieldValues());
                    }
                }, {
                    text: 'Cancel'
                }]
	        })]
	    });

	    this.callParent(arguments);
	}
});
